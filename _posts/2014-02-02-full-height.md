	window.onload = fullHeight;
window.onresize = fullHeight;

function fullHeight() {
  var h = window.innerHeight;
  var elem = document.querySelectorAll('.box-full');

  for (var i = 0; i < elem.length; i++) {
      elem[i].style.height = h +'px';
  }
}
